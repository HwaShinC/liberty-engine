/**
 * Provides Revision model.
 *
 * @module models
 * @submodule Revision
 */

'use strict';

const models = require('./');

/**
 * Model representing revisions.
 *
 * @class Revision
 */
module.exports = function(sequelize, DataTypes) {
  const Revision = sequelize.define('Revision', {
    id: {
      type: DataTypes.INTEGER,
      primaryKey: true,
      autoIncrement: true
    },
    changedLength: {
      type: DataTypes.INTEGER,
      field: 'changed_length',
      allowNull: false,
      defaultValue: 0
    },
    status: {
      type: DataTypes.ENUM('new', 'normal', 'moved', 'deleted')
    }
  }, {
    tableName: 'revision',
    underscored: true,
    classMethods: {
      associate: function(models) {
        Revision.belongsTo(models.Wikitext, {
          foreignKey: {
            name: 'wikitextId', field: 'wikitext_id', allowNull: false
          }
        });
        Revision.belongsTo(models.Article, {
          foreignKey: {
            name: 'articleId', field: 'article_id', allowNull: false
          }
        });
        Revision.belongsTo(models.User, {
          as: 'Author',
          foreignKey: {
            name: 'authorId', field: 'author_id', allowNull: false
          }
        });
      },

      /**
       * Create a new revision and make it latest revision of an article.
       * @method createNew
       * @static
       * @param {Object} option
       * @param {User} option.article an article to change.
       * @param {User} option.author user writing this.
       * @param {String} option.wikitext wikitext.
       * @param {String} option.status one of 'new', 'normal', 'moved', or 'deleted'.
       * @return {Promise<Revision>} Returns a promise of new revision.
       */
      createNew: function({ article, author, wikitext, status }) {
        return models.Wikitext.replaceOnSave({ article, author, wikitext, status })
        .then((replacedWikitext) => {
          if (status === 'new') {
            return models.Wikitext.create({ wikitext: replacedWikitext })
            .then((wikitext) => {
              return this.create({
                authorId: author.id,
                articleId: article.id,
                wikitextId: wikitext.id,
                changedLength: wikitext.length,
                status: status
              });
            })
            .then((revision) => {
              return article.addLatestRevision(revision);
            });
          }
        });
      }
    }
  });
  return Revision;
};
