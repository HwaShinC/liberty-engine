/**
 * Provides User model.
 *
 * @module models
 * @submodule User
 */

'use strict';

const bcrypt = require('bcrypt');
const saltRounds = 10;
const jwt = require('jsonwebtoken');
const env = process.env.NODE_ENV || 'development';
const secret = require('../config/config.json')[env].secret;
const models = require('./');

/**
 * Model representing users.
 *
 * @class User
 */
module.exports = function(sequelize, DataTypes) {
  const User = sequelize.define('User', {
    id: {
      type: DataTypes.INTEGER,
      primaryKey: true,
      autoIncrement: true
    },
    username: {
      type: DataTypes.STRING(128),
      allowNull: false,
      unique: true,
      validate: {
        notEmpty: true,
        len: [2, 128]
      }
    },
    passwordHash: {
      type: DataTypes.STRING(128),
      field: 'password_hash'
    },
    password: {
      type: DataTypes.VIRTUAL,
      validate: {
        len: [6, 128]
      }
    },
    email: {
      type: DataTypes.STRING(128),
      allowNull: true,
      isEmail: true,
      defaultValue: null
    }
  }, {
    tableName: 'user',
    underscored: true,
    classMethods: {
      /**
       * Describes association.
       * @method associate
       * @static
       * @param {Model[]} models
       */
      associate(models) {
        User.belongsToMany(models.UserGroup, { through: 'user_group_user' });
        User.hasMany(models.UserProperty, { foreignKey: 'userId' });
      },

      /**
       * Finds a user by username.
       * @method findByUsername
       * @static
       * @param {String} username
       * @return {Promise<User, null>} Returns a promise of user or null if not exists.
       */
      findByUsername(username) {
        return User.findOne({
          where: {
            username: username
          }
        });
      },

      /**
       * Verifies a json web token (jwt). If it valid, returns decoded data of it.
       * @method verifyToken
       * @static
       * @param {String} token jwt of a user.
       * @throws {Error} if the token is invalid.
       * @return {Promise<Object>} Returns a promise of decoded token data.
       */
      verifyToken(token) {
        return new Promise((resolve, reject) => {
          jwt.verify(token, secret, (err, decoded) => {
            if (err) {
              reject(err);
            } else {
              resolve(decoded);
            }
          });
        });
      }
    },
    instanceMethods: {
      /**
       * Returns whether given password is correct.
       * @method authenticate
       * @param {String} password password of this user
       * @throws {Error} if the token is invalid.
       * @return {Promise<Bool>} Returns a promise that has true (correct) or false (incorrect).
       */
      authenticate(password) {
        return bcrypt.compare(password, this.passwordHash);
      },

      /**
       * Issues a json web token (jwt). If it valid, returns decoded data of it.
       * @method issueToken
       * @param {String} token jwt of a user.
       * @throws {Error} if the token is invalid.
       * @return {Promise<Object>} Returns a promise of decoded token data.
       */
      issueToken() {
        return new Promise((resolve, reject) => {
          jwt.sign({
            id: this.id,
            username: this.username
          }, secret, { expiresIn: '1h' }, (err, token) => {
            if (err) {
              reject(err);
            } else {
              resolve(token);
            }
          });
        });
      },

      /**
       * Sets a property of this user.
       * @method setProperty
       * @param {String} name name of property.
       * @param {String} value value of property.
       * @return {Promise<UserProperty>} Returns a promise of new user property.
       */
      setProperty(name, value) {
        return models.UserProperty.upsert({
          userId: this.id,
          name, value
        });
      },

      /**
       * Returns specified property value.
       * @method getProperty
       * @param {String} name name of property.
       * @param {String} value value of property.
       * @return {Promise<String, null>} Returns a promise of new user property or null if not exists.
       */
      getProperty(name) {
        return this.getUserProperties({ where: { name } })
        .then(([property]) => {
          if (property) {
            return property.value;
          } else {
            return null;
          }
        });
      },

      /**
       * Returns signature of this user.
       * @method getSignature
       * @return {Promise<String>} Returns a promise of signature of this user.
       */
      getSignature() {
        return this.getProperty('signature')
        .then((signature) => {
          if (signature) {
            return this.getProperty('wikitext-sign')
            .then((wikitextsign) => {
              if (wikitextsign === null || wikitextsign === '0') {
                return `[[${this.userPageFullTitle}|${signature}]]`;
              } else {
                return signature;
              }
            });
          } else {
            return `[[${this.userPageFullTitle}]]`;
          }
        });
      }
    },
    getterMethods: {
      /**
       * Indicates the full title of userpage of this user.
       *
       * @attribute userPageFullTitle
       * @readOnly
       * @type String
       */
      userPageFullTitle() {
        return `사용자:${this.username}`;
      }
    },
    hooks: {
      beforeCreate: hashPassword,
      beforeUpdate: hashPassword
    }
  });
  return User;
};

function hashPassword(user) {
  if (!user.changed('password')) return;
  return bcrypt.hash(user.password, saltRounds)
  .then((hash) => {
    user.passwordHash = hash;
    user.password = undefined;
  });
}
